34 template <typename FnT>
35 std::istream& parse_section(std::istream& is, FnT fn, std::string* name = nullptr)
36 {
37  std::string line;
38  while (std::getline(is, line)) {
39 
40  trim(line);
41 
42  // Ignore empty lines and comments.
43 
44  if (line.empty() || line[0] == '#') {
45  continue;
46  }
47 
48  if (line.front() == '[' && line.back() == ']') {
49  if (name) {
50  name->assign(line, 1, line.size() - 2);
51  trim(*name);
52  }
53  return is;
54  }
55 
56  auto [key, val] = split_pair(line, '=');
57  rtrim(key);
58  ltrim(val);
59 
60  fn(key, val);
61  }
62  if (name) {
63  name->clear();
64  }
65  return is;
66 }
