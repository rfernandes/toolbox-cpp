59 void RateWindow::add(MonoTime time, size_t count) noexcept
60 {
61  const auto t = MonoClock::to_time_t(time);
62  auto& bucket = at(t);
63  // The fast path and simplest case is where the time falls within the same time bucket as the
64  // previous tick.
65  if (t == last_time_) {
66  count_ += count;
67  bucket += count;
68  return;
69  }
70  // Clock is assumed to be monotonic, so if the time is not the same, then it must be greater.
71  assert(t > last_time_);
72  // If the time has advanced by less than one complete cycle of the buffers.
73  if (t - last_time_ < static_cast<time_t>(buckets_.size())) {
74  // Advance past any interleaving buckets.
75  while (t != ++last_time_) {
76  auto& skipped = at(last_time_);
77  count_ -= skipped;
78  skipped = 0;
79  }
80  // And then adjust the count on the target bucket.
81  count_ = count_ + count - bucket;
82  } else {
83  // Zero the entire buffer.
84  fill(buckets_.begin(), buckets_.end(), 0);
85  count_ = count;
86  }
87  last_time_ = t;
88  bucket = count;
89 }
