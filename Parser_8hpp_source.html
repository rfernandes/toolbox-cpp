69  std::size_t parse(CyclTime now, ConstBuffer buf)
70  {
71  static http_parser_settings settings{make_settings()};
72  const auto rc = http_parser_execute(&parser_, &settings, buffer_cast<const char*>(buf),
73  buffer_size(buf));
74  const auto err = static_cast<http_errno>(parser_.http_errno);
75  if (err != HPE_OK) {
76  if (err == HPE_PAUSED) {
77  // Clear pause state.
78  http_parser_pause(&parser_, 0);
79  } else {
80  throw HttpException{HttpStatus::BadRequest,
81  err_msg() << http_errno_name(err) << ": "
82  << http_errno_description(err)};
83  }
84  }
85  return rc;
86  }

103  settings.on_chunk_complete = on_chunk_end;

122  {
