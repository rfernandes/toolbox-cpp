40 class HttpParser
41 : public BasicHttpParser<HttpParser>
42 , public BasicUrl<HttpParser> {
43  friend class BasicHttpParser<HttpParser>;
44  friend class BasicUrl<HttpParser>;
45 
46  public:
47  using BasicHttpParser<HttpParser>::BasicHttpParser;
48  ~HttpParser() = default;
49 
50  const auto& url() const noexcept { return url_; }
51  const auto& status() const noexcept { return status_; }
52  const auto& headers() const noexcept { return headers_; }
53  const auto& body() const noexcept { return body_; }
54 
55  void clear() noexcept
56  {
57  url_.clear();
58  status_.clear();
59  headers_.clear();
60  body_.clear();
61  }
62  using BasicHttpParser<HttpParser>::parse;
63 
64  private:
65  bool on_message_begin(CyclTime now) noexcept
66  {
67  BasicUrl<HttpParser>::reset();
68  clear();
69  return true;
70  }
71  bool on_url(CyclTime now, string_view sv) noexcept
72  {
73  url_.append(sv.data(), sv.size());
74  return true;
75  }
76  bool on_status(CyclTime now, string_view sv) noexcept
77  {
78  status_.append(sv.data(), sv.size());
79  return true;
80  }
81  bool on_header_field(CyclTime now, string_view sv, First first) noexcept
82  {
83  if (first == First::Yes) {
84  headers_.emplace_back(string{sv.data(), sv.size()}, "");
85  } else {
86  headers_.back().first.append(sv.data(), sv.size());
87  }
88  return true;
89  }
90  bool on_header_value(CyclTime now, string_view sv, First first) noexcept
91  {
92  headers_.back().second.append(sv.data(), sv.size());
93  return true;
94  }
95  bool on_headers_end(CyclTime now) noexcept { return true; }
96  bool on_body(CyclTime now, string_view sv) noexcept
97  {
98  body_.append(sv.data(), sv.size());
99  return true;
100  }
101  bool on_message_end(CyclTime now) noexcept
102  {
103  bool ret{false};
104  try {
105  if (!url_.empty()) {
106  BasicUrl<HttpParser>::parse();
107  }
108  ret = true;
109  } catch (const std::exception& e) {
110  cerr << "exception: " << e.what() << endl;
111  }
112  return ret;
113  }
114  bool on_chunk_header(CyclTime now, size_t len) noexcept { return true; }
115  bool on_chunk_end(CyclTime now) noexcept { return true; }
116 
117  string url_;
118  string status_;
119  vector<pair<string, string>> headers_;
120  string body_;
121 };
