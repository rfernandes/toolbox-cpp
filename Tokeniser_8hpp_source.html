42  void reset(std::string_view buf, std::string_view delims) noexcept
43  {
44  buf_ = buf;
45  delims_ = delims;
46  i_ = buf_.cbegin();
47  j_ = std::find_first_of(i_, buf_.cend(), delims_.cbegin(), delims_.cend());
48  }
50  std::size_t consumed() const noexcept { return i_ - buf_.cbegin(); }
52  bool empty() const noexcept { return i_ == buf_.cend(); }
54  bool has_delim() const noexcept { return j_ != buf_.cend(); }
55  std::string_view top() const noexcept { return buf_.substr(i_ - buf_.cbegin(), j_ - i_); }
56  std::string_view next() noexcept
57  {
58  const auto tok = top();
59  pop();
60  return tok;
61  }
62  void pop() noexcept
63  {
64  if (j_ != buf_.cend()) {
65  i_ = j_ + 1;
66  j_ = std::find_first_of(i_, buf_.cend(), delims_.cbegin(), delims_.cend());
67  } else {
68  i_ = j_;
69  }
70  }
71 
72  private:
73  std::string_view buf_;
74  std::string_view delims_;
75  // Assumption: std::string_view iterators are equivalent across copies of the std::string_view.
76  std::string_view::const_iterator i_, j_;
77 };
