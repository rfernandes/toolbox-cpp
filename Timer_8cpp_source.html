80 int TimerQueue::dispatch(CyclTime now)
81 {
82  int work{};
83  while (!heap_.empty()) {
84 
85  // If not pending, then must have been cancelled.
86  if (!heap_.front().pending()) {
87  pop();
88  --cancelled_;
89  assert(cancelled_ >= 0);
90  } else if (heap_.front().expiry() <= now.mono_time()) {

92  ++work;
93  } else {

95  }
96  }

129 void TimerQueue::expire(CyclTime now)
130 {
131  // Pop timer.
132  auto tmr = pop();
133  assert(tmr.pending());
134  try {
135  // Notify user.
136  tmr.slot().invoke(now, tmr);
137  } catch (const std::exception& e) {
138  TOOLBOX_ERROR << "error handling timer event: " << e.what();
139  }
140 
141  // If timer was not cancelled during the callback.
142  if (tmr.pending()) {
143 
144  // If periodic timer.
145  if (tmr.interval().count() > 0) {
146 
147  // Add interval to expiry, while ensuring that next expiry is always in the future.
148  tmr.set_expiry(max(tmr.expiry() + tmr.interval(), now.mono_time() + 1ns));
149 
150  // Reschedule popped timer.
151  heap_.push_back(tmr);
152  push_heap(heap_.begin(), heap_.end(), is_after);
153 
154  } else {
155 
156  // Free handler for non-repeating timer.
157  tmr.slot().reset();
158  }
159  }
160 }

162 void TimerQueue::gc() noexcept
163 {
164  // Garbage collect if more than half of the timers have been cancelled.
165  if (cancelled_ > static_cast<int>(heap_.size() >> 1)) {
166  const auto it
167  = remove_if(heap_.begin(), heap_.end(), [](const auto& tmr) { return !tmr.pending(); });
168  heap_.erase(it, heap_.end());
169  make_heap(heap_.begin(), heap_.end(), is_after);
170  cancelled_ = 0;
171  }

182 void intrusive_ptr_release(Timer::Impl* impl) noexcept
183 {
184  --impl->ref_count;
185  if (impl->ref_count == 1) {
186  // Cancel pending if only one reference remains. If only one reference remains after
187  // decrementing the counter, and the timer is still pending, then the final reference must
188  // be held within the internal timer queue, which means that no more references exist
189  // outside of the timer queue.
190  if (impl->slot) {
191  impl->slot.reset();
192  impl->tq->cancel();
193  }
194  } else if (impl->ref_count == 0) {
195  impl->tq->pool_.dealloc(impl);
196  }
197 }
