24 int Resolver::run(Duration timeout)
25 {
26  constexpr size_t BatchSize = 16;
27 
28  Lock lock{mutex_};
29  // Predicate returns â€‹false if the waiting should be continued.
30  const auto pred = [this] { return !this->queue_.empty() || stop_; };
31  // Returns false if predicate was false after timeout.
32  if (!cond_.wait_for(lock, timeout, pred)) {
33  return 0;
34  } else if (stop_) {
35  // This will unblock waiters by throwing a "broken promise" exception.
36  queue_.clear();
37  return -1;
38  }
39  // Copy batch of tasks to temporary buffer.
40  std::array<Task, BatchSize> tasks{};
41 
42  const auto n = std::min(queue_.size(), BatchSize);
43  for (std::size_t i{0}; i < n; ++i) {
44  // Copy is disabled for unique_ptr<>, so use swap() instead.
45  queue_[i].swap(tasks[i]);
46  }
47  queue_.erase(queue_.begin(), queue_.begin() + n);
48  lock.unlock();
49 
50  // Execute tasks after lock is released.
51  for (std::size_t i{0}; i < n; ++i) {
52  tasks[i]();
53  }
54  return n;
55 }
