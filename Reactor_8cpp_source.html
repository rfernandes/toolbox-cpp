55 int Reactor::poll(CyclTime now, Duration timeout)
56 {
57  enum { High = 0, Low = 1 };
58  using namespace chrono;
59 
60  // If timeout is zero then the wait_until time should also be zero to signify no wait.
61  MonoTime wait_until{};
62  if (!is_zero(timeout) && hooks_.empty()) {
63  const MonoTime next
64  = next_expiry(timeout == NoTimeout ? MonoClock::max() : now.mono_time() + timeout);
65  if (next > now.mono_time()) {
66  wait_until = next;
67  }
68  }
69  // TODO: consider using a dynamic buffer that scales with increased demand.
70  Event buf[MaxEvents];
71 
72  int n;
73  error_code ec;
74  if (wait_until < MonoClock::max()) {
75  // The wait function will not block if time is zero.
76  n = epoll_.wait(buf, MaxEvents, wait_until, ec);
77  } else {
78  // Block indefinitely.
79  n = epoll_.wait(buf, MaxEvents, ec);
80  }
81  if (ec) {
82  if (ec.value() != EINTR) {
83  throw system_error{ec};
84  }
85  return 0;
86  }
87  // If the epoller call was a blocking call, then acquire the current time.
88  if (!is_zero(wait_until)) {
89  now = CyclTime::now();
90  }
91  n = tqs_[High].dispatch(now) + dispatch(now, buf, n);
92  // Low priority timers are only dispatched during empty cycles.
93  if (n == 0) {
94  n += tqs_[Low].dispatch(now);
95  }
96  io::dispatch(now, hooks_);
97  return n;
98 }

107 MonoTime Reactor::next_expiry(MonoTime next) const
108 {
109  enum { High = 0, Low = 1 };
110  using namespace chrono;
111  {
112  auto& tq = tqs_[High];
113  if (!tq.empty()) {
114  // Duration until next expiry. Mitigate scheduler latency by preempting the
115  // high-priority timer and busy-waiting for 200us ahead of timer expiry.
116  next = min(next, tq.front().expiry() - 200us);
117  }
118  }
119  {
120  auto& tq = tqs_[Low];
121  if (!tq.empty()) {
122  // Duration until next expiry.
123  next = min(next, tq.front().expiry());
124  }
125  }
126  return next;
127 }

129 int Reactor::dispatch(CyclTime now, Event* buf, int size)
130 {
131  int work{0};
132  for (int i{0}; i < size; ++i) {
133 
134  auto& ev = buf[i];
135  const auto fd = epoll_.fd(ev);
136  if (fd == notify_.fd()) {
137  notify_.read();
138  continue;
139  }
140  const auto& ref = data_[fd];
141  if (!ref.slot) {
142  // Ignore timerfd.
143  continue;
144  }
145 
146  const auto sid = epoll_.sid(ev);
147  // Skip this socket if it was modified after the call to wait().
148  if (ref.sid > sid) {
149  continue;
150  }
151  // Apply the interest events to filter-out any events that the user may have removed from
152  // the events since the call to wait() was made. This would typically happen via a reentrant
153  // call into the reactor from an event-handler. N.B. EpollErr and EpollHup are always
154  // reported if they occur, regardless of whether they are specified in events.
155  const auto events = ev.events & (ref.events | EpollErr | EpollHup);
156  if (!events) {
157  continue;
158  }
159 
160  try {
161  ref.slot(now, fd, events);
162  } catch (const std::exception& e) {
163  TOOLBOX_ERROR << "error handling io event: " << e.what();
164  }
165  ++work;
166  }
167  return work;
168 }
