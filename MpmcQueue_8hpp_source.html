40 template <typename ValueT>
41 class MpmcQueue {
42  static_assert(std::is_trivially_copyable_v<ValueT>);
43 
44  public:
45  struct alignas(CacheLineSize) Elem {
46  std::int64_t seq;
47  ValueT val;
48  };
49  static_assert(std::is_trivially_copyable_v<Elem>);
50  struct alignas(CacheLineSize) Impl {
51  // Ensure that read and write positions are in different cache-lines.
52  std::int64_t rpos;
53  alignas(CacheLineSize) std::int64_t wpos;
54  alignas(CacheLineSize) Elem elems[];
55  };
56  static_assert(std::is_trivially_copyable_v<Impl>);
57  static_assert(sizeof(Impl) == 2 * CacheLineSize);
58  static_assert(offsetof(Impl, rpos) == 0 * CacheLineSize);
59  static_assert(offsetof(Impl, wpos) == 1 * CacheLineSize);
60  static_assert(offsetof(Impl, elems) == 2 * CacheLineSize);

62  constexpr MpmcQueue(std::nullptr_t = nullptr) noexcept {}
63  explicit MpmcQueue(std::size_t capacity)
64  : capacity_{next_pow2(capacity)}
65  , mask_{capacity_ - 1}
66  , mem_map_{os::mmap(nullptr, size(capacity_), PROT_READ | PROT_WRITE, MAP_ANON | MAP_PRIVATE,
67  -1, 0)}
68  , impl_{static_cast<Impl*>(mem_map_.get().data())}
69  {
70  assert(capacity >= 2);
71 
72  std::memset(impl_, 0, size(capacity_));
73  // Initialise sequence numbers.
74  for (std::int64_t i{0}; i < static_cast<std::int64_t>(capacity); ++i) {
75  __atomic_store_n(&impl_->elems[i].seq, i, __ATOMIC_RELAXED);
76  }
77  }

170  template <typename FnT>
171  bool read(FnT fn) noexcept
172  {
173  static_assert(std::is_nothrow_invocable_v<FnT, ValueT&&>);
174  auto rpos = __atomic_load_n(&impl_->rpos, __ATOMIC_RELAXED);
175  for (;;) {
176  auto& elem = impl_->elems[rpos & mask_];
177  const auto seq = __atomic_load_n(&elem.seq, __ATOMIC_ACQUIRE);
178  const auto diff = seq - (rpos + 1);
179  if (diff == 0) {
180  // The compare_exchange_weak function re-reads rpos on failure.
181  if (__atomic_compare_exchange_n(&impl_->rpos, &rpos, rpos + 1, true,
182  __ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
183  fn(std::move(elem.val));
184  // Commit.
185  __atomic_store_n(&elem.seq, rpos + capacity_, __ATOMIC_RELEASE);
186  break;
187  }
188  // Continue.
189  } else if (diff < 0) {
190  return false;
191  } else {
192  rpos = __atomic_load_n(&impl_->rpos, __ATOMIC_RELAXED);
193  }
194  }
195  return true;
196  }
198  template <typename FnT>
199  bool write(FnT fn) noexcept
200  {
201  static_assert(std::is_nothrow_invocable_v<FnT, ValueT&>);
202  auto wpos = __atomic_load_n(&impl_->wpos, __ATOMIC_RELAXED);
203  for (;;) {
204  auto& elem = impl_->elems[wpos & mask_];
205  const auto seq = __atomic_load_n(&elem.seq, __ATOMIC_ACQUIRE);
206  const auto diff = seq - wpos;
207  if (diff == 0) {
208  // The compare_exchange_weak function re-reads wpos on failure.
209  if (__atomic_compare_exchange_n(&impl_->wpos, &wpos, wpos + 1, true,
210  __ATOMIC_RELAXED, __ATOMIC_RELAXED)) {
211  fn(elem.val);
212  // Commit.
213  __atomic_store_n(&elem.seq, wpos + 1, __ATOMIC_RELEASE);
214  break;
215  }
216  // Continue.
217  } else if (diff < 0) {
218  return false;
219  } else {
220  wpos = __atomic_load_n(&impl_->wpos, __ATOMIC_RELAXED);
221  }
222  }
223  return true;
224  }
226  bool pop(ValueT& val) noexcept
227  {
228  static_assert(std::is_nothrow_move_assignable_v<ValueT>);
229  return read([&val](ValueT&& ref) noexcept { val = std::move(ref); });
230  }
232  bool push(const ValueT& val) noexcept
233  {
234  static_assert(std::is_nothrow_copy_assignable_v<ValueT>);
235  return write([&val](ValueT& ref) noexcept { ref = val; });
236  }
238  bool push(ValueT&& val) noexcept
239  {
240  static_assert(std::is_nothrow_move_assignable_v<ValueT>);
241  return write([&val](ValueT& ref) noexcept { ref = std::move(val); });
242  }

244  private:
245  static constexpr std::size_t capacity(std::size_t size) noexcept
246  {
247  return (size - sizeof(Impl)) / sizeof(Elem);
248  }
249  static constexpr std::size_t size(std::size_t capacity) noexcept
250  {
251  return sizeof(Impl) + capacity * sizeof(Elem);
252  }
253 
254  std::uint64_t capacity_{}, mask_{};
255  Mmap mem_map_{nullptr};
256  Impl* impl_{nullptr};
257 };

260 template <typename ValueT>
261 void new_mpmc_queue(FileHandle& fh, std::size_t capacity)
262 {
263  using Elem = typename MpmcQueue<ValueT>::Elem;
264  using Impl = typename MpmcQueue<ValueT>::Impl;
265 
266  assert(capacity >= 2);
267 
268  capacity = next_pow2(capacity);
269  const auto size = sizeof(Impl) + capacity * sizeof(Elem);
270 
271  os::ftruncate(fh.get(), size);
272  Mmap mem_map{os::mmap(nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, fh.get(), 0)};
273  auto* const impl = static_cast<Impl*>(mem_map.get().data());
274 
275  std::memset(impl, 0, size);
276  // Initialise sequence numbers.
277  for (std::int64_t i{0}; i < static_cast<std::int64_t>(capacity); ++i) {
278  __atomic_store_n(&impl->elems[i].seq, i, __ATOMIC_RELAXED);
279  }
280 }
