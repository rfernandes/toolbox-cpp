50  void run(ostream& os, const string& regex_str, bool randomise)
51  {
52  vector<Benchmark*> filtered;
53 
54  regex regex{regex_str};
55  for (const auto& bm : store_) {
56  if (regex_search(bm.first, regex)) {
57  filtered.push_back(bm.second);
58  }
59  }
60 
61  if (randomise) {
62  default_random_engine rng{};
63  shuffle(begin(filtered), end(filtered), rng);
64  }
65 
66  if (!filtered.empty()) {
67  BenchmarkSuite suite{os, 1000.0};
68  for (auto* bm : filtered) {
69  suite.run(bm->name, bm->fn);
70  }
71  }
72  }
73  void store(const char* name, Benchmark& bm) { store_.insert_or_assign(name, &bm); }
74 
75  private:

77 
78  map<const char*, Benchmark*> store_;
