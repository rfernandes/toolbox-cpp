43  /*
44  * Returns true if connection was established synchronously or false if connection is pending
45  * asynchronous completion.
46  */
47  bool connect(CyclTime now, Reactor& r, const Endpoint& ep)
48  {
49  StreamSockClnt sock{ep.protocol()};
50  static_cast<DerivedT*>(this)->on_sock_prepare(now, sock);
51  sock.set_non_block();
52  if (sock.is_ip_family()) {
53  set_tcp_no_delay(sock.get(), true);
54  }
55 
56  std::error_code ec;
57  sock.connect(ep, ec);
58  if (ec) {
59  if (ec != std::errc::operation_in_progress) {
60  throw std::system_error{ec, "connect"};
61  }
62  sub_
63  = r.subscribe(*sock, EpollIn | EpollOut, bind<&StreamConnector::on_io_event>(this));
64  ep_ = ep;
65  sock_ = std::move(sock);
66  return false;
67  }
68  static_cast<DerivedT*>(this)->on_sock_connect(now, std::move(sock), ep);
69  return true;
70  }
71 
72  protected:
73  ~StreamConnector() = default;
74 
75  private:
76  void on_io_event(CyclTime now, int fd, unsigned events)
77  {
78  IoSock sock{std::move(sock_)};
79  sub_.reset();
80  try {
81  const auto ec = sock.get_error();
82  if (ec) {
83  throw std::system_error{ec, "connect"};
84  }
85  static_cast<DerivedT*>(this)->on_sock_connect(now, std::move(sock), ep_);
86  } catch (const std::exception& e) {
87  static_cast<DerivedT*>(this)->on_sock_connect_error(now, e);
88  }
89  }
