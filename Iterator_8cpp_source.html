117 bool HdrPercentileIterator::do_next() noexcept
118 {
119  if (!has_next()) {
120  if (seen_last_value_) {
121  return false;
122  }
123  seen_last_value_ = true;
124  percentile_ = 100.0;
125  return true;
126  }
127 
128  if (counts_index_ == -1 && !basic_next()) {
129  return false;
130  }
131  do {
132  const double current_percentile{cumulative_count_ * 100.0 / h_.total_count()};
133  if (count_ != 0 && percentile_to_iterate_to_ <= current_percentile) {
134  update_iterated_values(h_.highest_equivalent_value(value_));
135 
136  percentile_ = percentile_to_iterate_to_;
137  const int64_t temp = (log(100.0 / (100.0 - percentile_to_iterate_to_)) / log(2)) + 1.0;
138  const int64_t half_distance = pow(2, static_cast<double>(temp));
139  const int64_t percentile_reporting_ticks{ticks_per_half_distance_ * half_distance};
140  percentile_to_iterate_to_ += 100.0 / percentile_reporting_ticks;
141 
142  return true;
143  }
144  } while (basic_next());
145 
146  return true;
147 }
