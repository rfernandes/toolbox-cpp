27 void run_reactor(Reactor& r, ThreadConfig config, const std::atomic<bool>& stop)
28 {
29  sig_block_all();
30  try {
31  set_thread_attrs(config);
32  TOOLBOX_NOTICE << "started " << config.name << " thread";
33  long i{0};
34  while (!stop.load(std::memory_order_acquire)) {
35  // Busy-wait for a small number of cycles after work was done.
36  constexpr long BusyCycles{100};
37  if (r.poll(CyclTime::now(), i++ < BusyCycles ? 0s : NoTimeout) > 0) {
38  // Reset counter when work has been done.
39  i = 0;
40  }
41  }
42  } catch (const std::exception& e) {
43  TOOLBOX_CRIT << "exception: " << e.what();
44  kill(getpid(), SIGTERM);
45  }
46  TOOLBOX_NOTICE << "stopping " << config.name << " thread";
47 }
48 } // namespace
