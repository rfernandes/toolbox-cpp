105 void std_logger(int level, string_view msg) noexcept
106 {
107  const auto now = WallClock::now();
108  const auto t = WallClock::to_time_t(now);
109  const auto ms = ms_since_epoch(now);
110 
111  struct tm tm;
112  localtime_r(&t, &tm);
113 
114  // The following format has an upper-bound of 42 characters:
115  // "%b %d %H:%M:%S.%03d %-7s [%d]: "
116  //
117  // Example:
118  // Mar 14 00:00:00.000 WARNING [0123456789]: msg...
119  // <---------------------------------------->
120  char head[42 + 1];
121  size_t hlen = strftime(head, sizeof(head), "%b %d %H:%M:%S", &tm);
122  hlen += sprintf(head + hlen, ".%03d %-7s [%d]: ", static_cast<int>(ms % 1000), log_label(level),
123  static_cast<int>(gettid()));
124  char tail{'\n'};
125  iovec iov[] = {
126  {head, hlen}, //
127  {const_cast<char*>(msg.data()), msg.size()}, //
128  {&tail, 1} //
129  };
130 
131  int fd{level > Log::Error ? STDOUT_FILENO : STDERR_FILENO};
132  // The following lock was required to avoid interleaving.
133  lock_guard<mutex> lock{mutex_};
134  // Best effort given that this is the logger.
135 #pragma GCC diagnostic push
136 #pragma GCC diagnostic ignored "-Wunused-result"
137  writev(fd, iov, sizeof(iov) / sizeof(iov[0]));
138 #pragma GCC diagnostic pop
139 }

141 void sys_logger(int level, string_view msg) noexcept
142 {
143  int prio;
144  switch (level) {
145  case Log::Crit:
146  prio = LOG_CRIT;
147  break;
148  case Log::Error:

150  break;
151  case Log::Warning:
152  prio = LOG_WARNING;
153  break;
154  case Log::Notice:
155  prio = LOG_NOTICE;
156  break;
157  case Log::Info:
158  prio = LOG_INFO;
159  break;
160  default:
161  prio = LOG_DEBUG;
162  }
163  syslog(prio, "%.*s", static_cast<int>(msg.size()), msg.data());
164 }
